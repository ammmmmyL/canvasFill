Join Effect

// object structure
const joinImage = {
     CreateCircleInMiddle,
     fontStyle: "Helvetica",
     fontColor: "black",
     backgroundColor: "#ffffff"
     circleColor: "#99ced3",
}
// calling function
joinImage.backgroundColor = "#D5A5A7"
joinImage.circleColor = "#a5bdd6dd"
const circleJoin1 = joinImage.CreateCircleInMiddle(img, caption, width=700, height=400, orientation="vertical")
// calling function
joinImage.backgroundColor = "#5f6366dd"
joinImage.circleColor = "#D5A5A7cd"
const circleJoin2 = joinImage.CreateCircleInMiddle(img, caption, width=400, height=700, orientation="horizontal")
